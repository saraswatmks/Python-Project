    numeric_features = ['price', 'mileage', 'year', 'ad_update_count', 'fake_mileage', 'unique_buyer_chat_sent_count',

    )

        'clf__colsample_bytree': (.7,.8,.9)
    }

    model = Pipeline([
        ('features', features),
        ('clf', clf)
    ]) 
    return model
